\section{Where is the cobordism bro}

\begin{theorem}[13.1]
    \label{thm:13-1}
    Let $f\colon M\to C$ be a smooth singular (?) $n$-manifold
    with bordism class $[M,f]\in\tilde\Omega_*(C)$.
    Then there exists a $PL$ manifold $\Sigma$
    such that the Hurewicz image of the cobordism class of $\Sigma$
    is congruent modulo the ideal generated by $\tilde H_*(MSO)$
    (in what ring?)
    to the image of $[M,f]$ under the composite
    \begin{equation*}
        \tilde\Omega_*(C)\overset h\to \tilde H_*(C)
	\overset{\sigma_*}\to H_*(BC)\subseteq H_*(BPL)
	\overset\phi\to H_*(MSPL).
    \end{equation*}
\end{theorem}
